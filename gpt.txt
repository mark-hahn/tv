
function getTvdbCharacters(tvdbData) {
  const characters = tvdbData?.response?.data?.characters;
  if (!characters || !Array.isArray(characters)) {
    return [];
  }
  return characters
    .filter(char => char.peopleType === 'Actor')
    .map(char => ({
      character:  char.name,
      actor:      char.personName,
      image:      char.personImgURL,
      tvdbUrl:    char.url,
      sortOrder:  char.sort,
      isFeatured: char.isFeatured
    }));
}
